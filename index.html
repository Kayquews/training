            <button onClick={decrementar}>-{step}</button>
            <button onClick={resetar}>Reset</button>
            <button onClick={incrementar}>+{step}</button>
        </div>
    );
}

from django.urls import reverse

class Categoria(models.Model):
    nome = models.CharField(max_length=100, unique=True)
    slug = models.SlugField(unique=True)
    ativa = models.BooleanField(default=True)
    
    class Meta:
        verbose_name_plural = "Categorias"

        return self.nome

class Post(models.Model):
    titulo = models.CharField(max_length=200)
    slug = models.SlugField(unique=True)
    conteudo = models.TextField()
    autor = models.ForeignKey(User, on_delete=models.CASCADE)
    categoria = models.ForeignKey(Categoria, on_delete=models.SET_NULL, null=True)
    data_criacao = models.DateTimeField(auto_now_add=True)
    data_atualizacao = models.DateTimeField(auto_now=True)
    publicado = models.BooleanField(default=False)
    
    class Meta:
        ordering = ['-data_criacao']
    
    def __str__(self):
        return self.titulo
    
    def get_absolute_url(self):
        return reverse('post-detail', kwargs={'slug': self.slug})

-- Joins complexos e subqueries
SELECT 
    u.nome AS usuario,
    u.email,
    COUNT(p.id) AS total_posts,
    AVG(p.visualizacoes) AS media_visualizacoes,
    MAX(p.data_criacao) AS ultimo_post,
    c.nome AS categoria_mais_usada
FROM usuarios u
LEFT JOIN posts p ON u.id = p.autor_id
LEFT JOIN categorias c ON p.categoria_id = c.id
WHERE u.ativo = true
  AND p.data_criacao >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY u.id, u.nome, u.email, c.nome
HAVING COUNT(p.id) > 0
ORDER BY total_posts DESC, media_visualizacoes DESC
LIMIT 10;

SELECT 
    p.titulo,
    p.data_criacao,
    (
        SELECT COUNT(*) 
        FROM comentarios c 
        WHERE c.post_id = p.id 
        AND c.aprovado = true
    ) AS total_comentarios,
    CASE 
        WHEN p.data_criacao >= CURRENT_DATE - INTERVAL '7 days' THEN 'Novo'

          email: "$autor.email"
        }
      },

        $push: {
          $cond: [
            { $gte: ["$likes", 100] },
            {
              titulo: "$titulo",
              likes: "$likes",
              data: "$data_criacao"
            },
            "$$REMOVE"
          ]
        }
      }
    }
  },
  
  // Stage 5: Sort - ordenar por total de posts
  {
    $sort: { total_posts: -1 }
  },
  
  // Stage 6: Project - formatar sa√≠da
  {
    $project: {
      categoria: "$_id",
      estatisticas: {
        total_posts: 1,
        media_likes: { $round: ["$media_likes", 2] },
        total_visualizacoes: 1
      },
      total_autores: { $size: "$autores" },
      posts_populares: { $slice: ["$posts_populares", 5] }
    }
  }
